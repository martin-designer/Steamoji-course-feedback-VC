<script>

  const gems = document.querySelectorAll('.gem');

  let selectedFeedback = '';


  gems.forEach(gem => {

    gem.addEventListener('click', () => {

      gems.forEach(g => g.classList.remove('selected'));

      gem.classList.add('selected');


      switch(gem.id) {

        case 'black':

          selectedFeedback = 'Do not like';

          break;

        case 'grey':

          selectedFeedback = 'It is OK';

          break;

        case 'gold':

          selectedFeedback = 'Like';

          break;

      }

      document.getElementById('result').innerText = `You selected: ${selectedFeedback}`;

    });

  });


  document.getElementById('submitBtn').addEventListener('click', () => {

    if(selectedFeedback === '') {

      alert('Please select a gem first!');

    } else {

      alert(`Your feedback has been submitted: ${selectedFeedback}`);

      console.log(`Feedback submitted: ${selectedFeedback}`);

      // 可以扩展，把结果发送到数据库或 Google Sheet

    }

  });

</script>
